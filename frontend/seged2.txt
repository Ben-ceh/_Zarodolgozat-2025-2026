import { useState,useEffect } from "react"
import Cim from "../Cim"
import "../App.css"

const BejegyzesekOsszesen=()=>{
    const [adatok,setAdatok]=useState([])
    const [tolt,setTolt]=useState(true)
    const [hiba,setHiba]=useState(false)



    useEffect(()=>{
        
    const leToltes=async ()=>{
        try{
           
            const response=await fetch(Cim.Cim+"/bejegyEsFelh")
            const data=await response.json()
            //alert(JSON.stringify(data))
            //console.log(data)
            if (response.ok)
                {
                    setAdatok(data)
                    setTolt(false)}
            else 
                {
                    setHiba(true)
                    setTolt(false)
                }
            }
        catch (error){
            console.log(error)
            setHiba(true)
        }
        
    }

        leToltes()
    },[])
const formatDate = (mysqlDate) => {
  return mysqlDate.split("T")[0]; // "2025-11-17"
};

    if (tolt)
        return (
            <div style={{textAlign:"center"}}>Adatok betöltése folyamatban...</div>
                )
    else if (hiba)
        return (
            <div>Hiba</div>
                )       
    
    else return (
        <div>
                {/*kivalasztott*/}
                {/* {adatok.map((elem,index)=>( */}
                    
                {adatok.map((elem,index)=>(
                     
                    <div >
                       <div >
                            
                            </div >   
                          
                        <div class="card doboz" style={{width:"500px"}} >
                        {/* Utnak a kepe */}
                    
                            
                            {elem.kep_url!=null?
                            <img class="card-img-top" src={`${Cim.Cim}/bejegyzesKepek/${elem.kep_url}`} alt={elem.felhasznalo_id}/>
                            :<img class="card-img-top" src={`${Cim.Cim}/bejegyzesKepek/X.png`} alt={elem.felhasznalo_id}/>}

                        
                        {/* é */}
                        <div class="card-body ">
                            <h4 class="card-title">
                                 {/* profil kep */}
                                 {elem.profil_kep==null&&elem.neme===1?
                            <img align="left" className="profilKep"   src={`${Cim.Cim}/kepekFelhasznalo/M.jpg`} alt={elem.felhasznalo_id}/>
                            :null}
                            {elem.profil_kep==null&&elem.neme===2?
                            <img className="profilKep" src={`${Cim.Cim}/kepekFelhasznalo/F.jpg`} alt={elem.felhasznalo_id}/>
                            :null}
                            {elem.profil_kep!=null?
                            <img className="profilKep" src={`${Cim.Cim}/kepekFelhasznalo/${elem.profil_kep}`} alt={elem.felhasznalo_id}/>
                            :null}
                                {elem.felhasznalonev}</h4>
                                <h1 class="card-text">{elem.cim}</h1>
                            <p class="card-text">{elem.tartalom}</p>
                            <div>
                            <p  class="text-secondary" style={{textAlign:"right", fontWeight:"bold"}}>{formatDate(elem.letrehozva)} </p>
                            </div>
                            
                        </div>
                        </div>
                        <br />    
                    </div>
                ))}
           
        </div>
    )
}
export default BejegyzesekOsszesen
